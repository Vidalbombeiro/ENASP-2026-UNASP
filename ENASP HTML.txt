<script>
    // Lógica para calcular o valor total dinamicamente
    const radioCombos = document.querySelectorAll('input[name="combo"]');
    const checkProgramacao = document.getElementById('programacao');
    const spanTotal = document.getElementById('valorTotal');

    function calcularTotal() {
        let total = 0;
        
        radioCombos.forEach(radio => {
            if(radio.checked) {
                total += parseFloat(radio.getAttribute('data-preco'));
            }
        });

        if(checkProgramacao.checked) {
            total += parseFloat(checkProgramacao.getAttribute('data-preco'));
        }

        spanTotal.innerText = total.toFixed(2).replace('.', ',');
    }

    radioCombos.forEach(radio => radio.addEventListener('change', calcularTotal));
    checkProgramacao.addEventListener('change', calcularTotal);

    // Validação do formulário
    function validarCampo(id, errorId) {
        const campo = document.getElementById(id);
        const erro = document.getElementById(errorId);
        if (campo.value.trim() === '') {
            campo.classList.add('input-error');
            erro.style.display = 'block';
            return false;
        } else {
            campo.classList.remove('input-error');
            erro.style.display = 'none';
            return true;
        }
    }

    function validarCombo() {
        const comboError = document.getElementById('comboError');
        const selecionado = document.querySelector('input[name="combo"]:checked');
        if (!selecionado) {
            comboError.style.display = 'block';
            return false;
        } else {
            comboError.style.display = 'none';
            return true;
        }
    }

    function validarPagamento() {
        const pagamento = document.getElementById('pagamento');
        const erro = document.getElementById('pagamentoError');
        if (pagamento.value === '') {
            pagamento.classList.add('input-error');
            erro.style.display = 'block';
            return false;
        } else {
            pagamento.classList.remove('input-error');
            erro.style.display = 'none';
            return true;
        }
    }

    // Lógica de envio do formulário
    const form = document.getElementById('inscricaoForm');
    
    form.addEventListener('submit', async (e) => {
        e.preventDefault();
        
        // Validar todos os campos
        const validoNome = validarCampo('nome', 'nomeError');
        const validoRg = validarCampo('rg', 'rgError');
        const validoCpf = validarCampo('cpf', 'cpfError');
        const validoCidade = validarCampo('cidade', 'cidadeError');
        const validoEstado = validarCampo('estado', 'estadoError');
        const validoIgreja = validarCampo('igreja', 'igrejaError');
        const validoForca = validarCampo('forca', 'forcaError');
        const validoCombo = validarCombo();
        const validoPagamento = validarPagamento();

        if (!validoNome || !validoRg || !validoCpf || !validoCidade || !validoEstado || 
            !validoIgreja || !validoForca || !validoCombo || !validoPagamento) {
            // Rolar para o primeiro erro
            const primeiroErro = document.querySelector('.input-error, #comboError[style*="display: block"], #pagamentoError[style*="display: block"]');
            if (primeiroErro) {
                primeiroErro.scrollIntoView({ behavior: 'smooth', block: 'center' });
            }
            return;
        }

        const btn = document.getElementById('btnSubmit');
        const textoOriginal = btn.innerText;
        btn.innerText = "⏳ Enviando...";
        btn.disabled = true;

        const formData = new FormData(form);

        try {
            const resposta = await fetch(form.action, {
                method: 'POST',
                body: formData
            });

            // Esconde o formulário
            form.style.display = 'none';
            
            const mensagemSucesso = document.getElementById('successMessage');
            mensagemSucesso.style.display = 'block';
            
            // Mostra a tela de pagamento
            const telaPagamento = document.getElementById('telaPagamento');
            telaPagamento.style.display = 'block';

            // Passa o valor total calculado para a tela final
            document.getElementById('valorFinal').innerText = spanTotal.innerText;
            
            // Verifica a forma de pagamento e mostra a área correta
            const formaPagamento = document.getElementById('pagamento').value;
            if(formaPagamento === 'PIX') {
                document.getElementById('areaPix').style.display = 'block';
            } else if (formaPagamento === 'Cartão de Crédito') {
                document.getElementById('areaCartao').style.display = 'block';
            }

            // Rolar para o topo para mostrar a tela de pagamento
            window.scrollTo({ top: 0, behavior: 'smooth' });

        } catch (error) {
            console.error('Erro!', error);
            alert('Erro ao enviar os dados. Verifique sua conexão e tente novamente.');
            btn.innerText = textoOriginal;
            btn.disabled = false;
        }
    });

    // Função para copiar chave PIX
    function copiarChave() {
        const chave = 'seu-email@provedor.com';
        navigator.clipboard.writeText(chave).then(() => {
            alert('Chave PIX copiada para a área de transferência!');
        }).catch(() => {
            // Fallback para navegadores mais antigos
            const textarea = document.createElement('textarea');
            textarea.value = chave;
            document.body.appendChild(textarea);
            textarea.select();
            document.execCommand('copy');
            document.body.removeChild(textarea);
            alert('Chave PIX copiada para a área de transferência!');
        });
    }

    // Adicionar validação em tempo real nos campos
    document.querySelectorAll('input, select').forEach(campo => {
        campo.addEventListener('blur', function() {
            if (this.id === 'pagamento') {
                validarPagamento();
            } else if (this.id && this.id.endsWith('Error')) {
                // Ignorar mensagens de erro
            } else {
                const errorId = this.id + 'Error';
                const errorEl = document.getElementById(errorId);
                if (errorEl && this.hasAttribute('required')) {
                    validarCampo(this.id, errorId);
                }
            }
        });
    });

    // Validação de combo em tempo real
    document.querySelectorAll('input[name="combo"]').forEach(radio => {
        radio.addEventListener('change', function() {
            document.getElementById('comboError').style.display = 'none';
        });
    });

    // Inicializar cálculo
    calcularTotal();
</script>
